  isActive: boolean;
  onPlay: () => void;

  const player = useAudioPlayer(item.audio);
  const status = useAudioPlayerStatus(player);
  const isPlaying = status.playing;

    if (isPlaying) {
      player.pause();
    } else {
      player.seekTo(0);
      player.play();
      onPlay();
